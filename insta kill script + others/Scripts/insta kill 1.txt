local Players = game:GetService("Players")
local player = Players.LocalPlayer
local character = player.Character or player.CharacterAdded:Wait()
local tool = character:WaitForChild("StunStick")

-- Function to instantly kill NPCs upon each hit
local function instantKill(hit)
    local humanoid = hit.Parent:FindFirstChildOfClass("Humanoid")
    if humanoid and hit.Parent ~= character then
        humanoid.Health = 0  -- Set NPC's health to zero to instantly kill them
    end
end

-- Connect the function to the tool's handle touch event
tool.Handle.Touched:Connect(instantKill)

-- Ensure the tool is always in the player's backpack
player.Backpack.ChildAdded:Connect(function(child)
    if child:IsA("Tool") and child.Name == "StunStick" then
        child.Handle.Touched:Connect(instantKill)
    end
end)

-- Also ensure the tool in character is connected after respawn
player.CharacterAdded:Connect(function(char)
    local stunStick = char:WaitForChild("StunStick")
    stunStick.Handle.Touched:Connect(instantKill)
end)
